function generateReport() { const phone = "4917623638794"; // Almanya formatı (0176... -> 49176...) const anlage = document.getElementById("anlage").value || "Nicht angegeben"; const stoerung = document.getElementById("stoerung").value || "Keine"; const dauer = document.getElementById("dauer").value || "0"; const notiz = document.getElementById("notiz").value || "-"; // 1. Personel Listesi let workers = []; document.querySelectorAll(".worker-select").forEach(select => { if (select.value) workers.push(select.value); }); // 2. Artikel Listesi let artikelText = ""; document.querySelectorAll(".artikel-item").forEach((item, index) => { const nr = item.querySelector(".artikel-no").value || "N/A"; const gut = item.querySelector(".gut-sayisi").value || "0"; const aus = item.querySelector(".aus-sayisi").value || "0"; artikelText += `\n📦 *Art ${index + 1}:* ${nr}\n ✅ Gut: ${gut} | ❌ Aus: ${aus}`; }); // 3. Mesaj Formatı let message = `📊 *APG PRODUKTIONSBERICHT*\n`; message += `━━━━━━━━━━━━━━━━\n`; message += `🏗️ *Anlage:* ${anlage}\n`; message += `👥 *Personal:* ${workers.join(", ") || "Keine Angabe"}\n`; message += `━━━━━━━━━━━━━━━━\n`; message += `*PRODUKTION:*${artikelText}\n`; message += `━━━━━━━━━━━━━━━━\n`; message += `⚠️ *Störung:* ${stoerung} (${dauer} Min)\n`; message += `📝 *Notiz:* ${notiz}\n`; message += `📅 *Datum:* ${new Date().toLocaleDateString('de-DE')} | ${new Date().toLocaleTimeString('de-DE', {hour: '2-digit', minute:'2-digit'})}`; // 4. WhatsApp'a Gönder const waUrl = `https://wa.me/${phone}?text=${encodeURIComponent(message)}`; window.open(waUrl, '_blank'); }
